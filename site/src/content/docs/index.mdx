## Quick Example

```typescript
import { createServer, json } from "verb";

const app = createServer({ port: 3000 });

// Define routes
app.get("/", () => new Response("Hello from Verb!"));

app.get("/api/users", () => {
  return json([
    { id: 1, name: "Alice" },
    { id: 2, name: "Bob" }
  ]);
});

app.get("/api/users/:id", (req, params) => {
  return json({ id: params.id, name: `User ${params.id}` });
});

// Start the server
console.log("Server running at http://localhost:3000");
```
